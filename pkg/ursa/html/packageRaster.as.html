as.RasterR Documentation

Coercion to package 'raster' objects

Description

as.Raster converts singe-band rmapRaster object to raster, multi-band rmapRaster object to brick and list of rmapRaster objects to stack. S4 classes “raster”, “brick”, and “stack” are defined in package raster.

Usage

as.Raster(obj)

## S3 method for class 'rmapRaster'
as.Raster(obj)

## S3 method for class 'list'
as.Raster(obj)

## S3 method for class 'rmapStack'
as.Raster(obj)

Arguments

obj

Object of class rmapRaster or list of rmapRaster objects

Details

Package raster is required for conversions.

The uppercase as.\bold{R}aster is important, because as.raster is used in internal functions for coersion to object of class raster.

Single-banded rmapRaster object (with or without colortable) is coerced to RasterLayer. Colortables are kept.
Multi-banded rmapRaster object is coerced to RasterBrick. Colortables are destroyed.
Multi-layered object (list of rmapRaster objects) is coerced to RasterStack. Colortables are destroyed.

Value

Either RasterLayer, RasterBrick, or RasterStack object.

If package raster is not installed then return value is NULL

Note

Package raster is marked as "Suggested".

Examples

session.grid(NULL)
if (requireNamespace("raster")) {
   require(methods) ## need to 'raster::plot'
   msk <- rmap.dummy(1,min=0,max=100)>40
   a1 <- rmap.dummy(1,min=200,max=500)[msk]
   a2 <- colorize(a1,ramp=FALSE)
   a3 <- as.integer(rmap.dummy(3,min=0,max=255.99))
   a4 <- rmap.stack(a3[msk])
   if (isLayer <- TRUE) {
      print(a1)
      r1 <- as.Raster(a1)
      message(as.character(class(r1)))
      print(r1)
      raster::plot(r1)
      b1 <- as.rmap(r1)
      print(c(exported=a1,imported=b1,failed=b1-a1))
      print(c(theSameValue=identical(rmap.value(a1),rmap.value(b1))
             ,rheSameGrid=identical(rmap.grid(a1),rmap.grid(b1))))
   }
   if (isLayerColortable <- TRUE) {
      r2 <- as.Raster(a2)
      message(as.character(class(r2)))
      print(r2)
      raster::plot(r2)
      b2 <- as.rmap(r2)
      print(c(theSameValue=identical(rmap.value(a2),rmap.value(b2))
             ,rheSameGrid=identical(rmap.grid(a2),rmap.grid(b2))))
   }
   if (isBrickOrRGB <- TRUE) {
      r3 <- as.Raster(a3)
      message(as.character(class(r3)))
      print(r3)
      raster::plot(r3)
      raster::plotRGB(r3)
      b3 <- as.rmap(r3)
      print(c(theSameValue=identical(rmap.value(a3),rmap.value(b3))
             ,rheSameGrid=identical(rmap.grid(a3),rmap.grid(b3))))
   }
   if (isStack <- TRUE) {
      r4 <- as.Raster(a4)
      message(as.character(class(r4)))
      print(r4)
      raster::plot(r4)
      b4 <- as.rmap(r4)
      print(c(theSameValue=identical(rmap.value(a4),rmap.value(b4))
             ,rheSameGrid=identical(rmap.grid(a4),rmap.grid(b4))))
   }
}
